function runProgram(input){

let stack = {
  arr: [],
  top: -1,
 
  isEmpty: function() {
    return (this.top == -1)
  },

  size: function() {
    return (this.top + 1)
  },

  peek: function() {
    if (this.top == -1)
      return "stack is empty";

    return this.arr[this.top]
  },

  push: function(elem) {
    this.top++
    this.arr[this.top] = elem
  },

  pop: function() {
    if (this.top == -1)
      return "stack is empty";

    const temp = this.arr[this.top]
    this.top--
    return temp
  },
  
 }
 


 let newarr=""
 
 for(let i=0;i<size;i++){
        while(stack.peek()>=A[i])
          stack.pop()
        if(stack.isEmpty())
          newarr+=-1+" "
        else
          newarr+=stack.peek()+" "
          
      stack.push(A[i])
 }
 console.log(newarr)
