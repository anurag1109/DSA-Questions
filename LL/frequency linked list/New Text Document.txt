const LinkedListNode = class {
    constructor(nodeData, frequency) {
        this.data = nodeData;
        this.frequency = frequency;
        this.next = null;
    }
};
// Complete the function below
function frequencyLinkedList(arr, n){
    arr.sort((a,b)=>a-b)
    let obj={}
    for(let i=0;i<n;i++){
        if(obj[arr[i]]==undefined)
        obj[arr[i]]=1
        else
        obj[arr[i]]+=1
    }
    
    
    let head=null
    let temp=null
    for(let key in obj){
        const node=new LinkedListNode(key,obj[key])
        if(head==null){
        head=node
        temp=node
        }else{
        temp.next=node
        temp=node
        }
    }
    return head
}
