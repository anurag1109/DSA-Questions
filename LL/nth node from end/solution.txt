const LinkedListNode = class {
    constructor(nodeData) {
        this.data = nodeData;
        this.next = null;
    }
};

function nth_node(head,n){
    let i=0
    let temp=head
    let cnt=1
    while(temp.next!=null){
    temp=temp.next
    cnt++
}
i=0
temp=head
while(i!=cnt-n){
    temp=temp.next
    i++
}
return(temp.data)
}
